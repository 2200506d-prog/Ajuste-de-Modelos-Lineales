            <h3>Explicación del Procedimiento Matricial</h3>
            <p>Para encontrar la mejor recta $Y = mx + b$, utilizamos el método de <strong>Mínimos Cuadrados</strong> mediante álgebra lineal:</p>
            <ol>
                <li><strong>Matriz de Diseño (A):</strong> Construimos una matriz donde la primera columna son "1"s (para el intercepto $b$) y la segunda columna son los valores de $X$.</li>
                <li><strong>Vector de Observaciones (y):</strong> Es un vector columna con los valores de $Y$.</li>
                <li><strong>Ecuación Normal:</strong> Resolvemos el sistema $(A^T A) \hat{\beta} = A^T y$.</li>
                <li><strong>Cálculo:</strong> Multiplicamos la transpuesta de $A$ por $A$, calculamos su inversa y la multiplicamos por el producto de $A^T$ y $y$. Los resultados son los coeficientes óptimos que minimizan la distancia al cuadrado de los puntos a la recta.</li>
            </ol>

<script>
    let chartInstance = null;

    function showSection(id) {
        document.querySelectorAll('.input-section').forEach(s => s.style.display = 'none');
        document.getElementById(id).style.display = 'block';
    }

    // Lógica para Excel
    document.getElementById('upload-excel').addEventListener('change', function(e) {
        const reader = new FileReader();
        reader.onload = function(e) {
            const data = new Uint8Array(e.target.result);
            const workbook = XLSX.read(data, {type: 'array'});
            const sheet = workbook.Sheets[workbook.SheetNames[0]];
            const json = XLSX.utils.sheet_to_json(sheet, {header: 1});
            
            let xArr = [], yArr = [];
            json.forEach(row => {
                if(!isNaN(row[0]) && !isNaN(row[1])) {
                    xArr.push(parseFloat(row[0]));
                    yArr.push(parseFloat(row[1]));
                }
            });
            processData(xArr, yArr);
        };
        reader.readAsArrayBuffer(e.target.files[0]);
    });

    // Generar tabla manual
    function generateTable() {
        const n = document.getElementById('num-elements').value;
        let html = '<table><tr><th>X</th><th>Y</th></tr>';
        for(let i=0; i<n; i++) {
            html += `<tr>
                <td><input type="number" class="in-x" step="any"></td>
                <td><input type="number" class="in-y" step="any"></td>
            </tr>`;
        }
        html += '</table>';
        document.getElementById('table-container').innerHTML = html;
    }

    function calculateManual() {
        const xInputs = document.querySelectorAll('.in-x');
        const yInputs = document.querySelectorAll('.in-y');
        let xArr = [], yArr = [];
        xInputs.forEach((el, i) => {
            if(el.value !== "") {
                xArr.push(parseFloat(el.value));
                yArr.push(parseFloat(yInputs[i].value));
            }
        });
        processData(xArr, yArr);
    }

    // Cálculos Matemáticos (Matrices simplificadas)
    function processData(x, y) {
        const n = x.length;
        let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
        for(let i=0; i<n; i++) {
            sumX += x[i]; sumY += y[i];
            sumXY += x[i]*y[i]; sumXX += x[i]*x[i];
        }

        // Resolución de Cramer para el sistema de matrices 2x2
        // [ n     sumX ] [ b ] = [ sumY ]
        // [ sumX  sumXX] [ m ] = [ sumXY ]
        const det = (n * sumXX) - (sumX * sumX);
        const b = (sumY * sumXX - sumX * sumXY) / det;
        const m = (n * sumXY - sumX * sumY) / det;

        displayResults(x, y, m, b);
    }

    function displayResults(x, y, m, b) {
        document.getElementById('output').style.display = 'block';
        document.getElementById('equation-output').innerHTML = `
            <strong>RESULTADOS DEL MODELO:</strong><br>
            Ecuación: Y = ${m.toFixed(4)}x + ${b.toFixed(4)}<br>
            Pendiente (m): ${m.toFixed(4)}<br>
            Intercepto (b): ${b.toFixed(4)}
        `;

        if(chartInstance) chartInstance.destroy();

        const ctx = document.getElementById('regressionChart').getContext('2d');
        chartInstance = new Chart(ctx, {
            type: 'scatter',
            data: {
                datasets: [{
                    label: 'Datos',
                    data: x.map((v, i) => ({x: v, y: y[i]})),
                    backgroundColor: '#1a73e8'
                }, {
                    label: 'Ajuste Lineal',
                    data: [
                        {x: Math.min(...x), y: m*Math.min(...x)+b},
                        {x: Math.max(...x), y: m*Math.max(...x)+b}
                    ],
                    type: 'line',
                    borderColor: '#ea4335',
                    fill: false
                }]
            }
        });
    }
